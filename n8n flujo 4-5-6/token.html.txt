  <script>
    /*
      CAMBIA ESTAS URLS SI TUS WEBHOOKS EN N8N TIENEN OTRO PATH.

      Flujo 4 debe recibir multipart/form-data porque aqui se mandan imagenes.
      WEBHOOK_BUSCAR_CLIENTE puede regresar JSON con los campos del cliente.
    */
    const WEBHOOK_FLUJO_4 = 'https://n8n.lacabanaeventos.com/webhook/generar-link-seleccion';
    const WEBHOOK_BUSCAR_CLIENTE = 'https://n8n.lacabanaeventos.com/webhook/buscar-cliente-id';

    const form = document.getElementById('tokenForm');
    const statusBox = document.getElementById('status');
    const buscarBtn = document.getElementById('buscarBtn');
    const nuevoBtn = document.getElementById('nuevoBtn');
    const buscarInput = document.getElementById('cliente_id_buscar');
    const preview = document.getElementById('preview');
    const evidencias = document.getElementById('evidencias');

    const fields = [
      'cliente_id',
      'nombre',
      'telefono',
      'email',
      'fecha_evento',
      'tipo_evento',
      'paquete_asignado',
      'num_invitados_estimado',
      'servicio_adicional'
    ];

    const summary = {
      cliente_id: document.getElementById('sClienteId'),
      nombre: document.getElementById('sNombre'),
      telefono: document.getElementById('sTelefono'),
      email: document.getElementById('sEmail'),
      fecha_evento: document.getElementById('sFecha'),
      tipo_evento: document.getElementById('sEvento'),
      paquete_asignado: document.getElementById('sPaquete'),
      num_invitados_estimado: document.getElementById('sInvitados'),
      servicio_adicional: document.getElementById('sServicio'),
      imagenes: document.getElementById('sImagenes')
    };

    const showStatus = (type, text) => {
      statusBox.className = type ? `status ${type}` : 'status';
      statusBox.textContent = text || '';
    };

    const showGeneratedLink = (result) => {
      const link = result && result.link ? String(result.link) : '';
      const token = result && result.token ? String(result.token) : '';

      statusBox.className = 'status ok';
      statusBox.innerHTML = link
        ? `Listo. Token generado: ${token}. Link: <a href="${link}" target="_blank" rel="noopener">${link}</a>`
        : 'Listo. El Flujo 4 recibio los datos y genero el token.';
    };

    const updateSummary = () => {
      for (const name of fields) {
        summary[name].textContent = form.elements[name].value || '-';
      }
      const count = evidencias.files.length;
      summary.imagenes.textContent = count ? `${count} archivo(s)` : '-';
    };

    const renderPreview = () => {
      preview.innerHTML = '';
      Array.from(evidencias.files).slice(0, 6).forEach((file) => {
        const box = document.createElement('div');
        box.className = 'thumb';
        const img = document.createElement('img');
        img.alt = file.name;
        img.src = URL.createObjectURL(file);
        img.onload = () => URL.revokeObjectURL(img.src);
        box.appendChild(img);
        preview.appendChild(box);
      });
      updateSummary();
    };

    const fillForm = (data) => {
      for (const name of fields) {
        if (data[name] !== undefined && form.elements[name]) {
          form.elements[name].value = data[name] || '';
        }
      }
      if (data.observaciones !== undefined) form.elements.observaciones.value = data.observaciones || '';
      updateSummary();
    };

    form.addEventListener('input', updateSummary);
    evidencias.addEventListener('change', renderPreview);

    form.addEventListener('reset', () => {
      setTimeout(() => {
        preview.innerHTML = '';
        showStatus('', '');
        updateSummary();
      }, 0);
    });

    nuevoBtn.addEventListener('click', () => {
      form.reset();
      buscarInput.value = '';
      form.elements.cliente_id.focus();
    });

    buscarBtn.addEventListener('click', async () => {
      const id = buscarInput.value.trim();
      if (!id) {
        showStatus('bad', 'Escribe primero el ID interno del cliente.');
        return;
      }

      buscarBtn.disabled = true;
      buscarBtn.textContent = 'Buscando...';
      showStatus('', '');

      try {
        const response = await fetch(`${WEBHOOK_BUSCAR_CLIENTE}?cliente_id=${encodeURIComponent(id)}`, {
          method: 'GET'
        });

        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const data = await response.json();

        if (!data || data.found === false) {
          form.elements.cliente_id.value = id;
          updateSummary();
          showStatus('bad', 'No encontre ese ID. Puedes capturarlo como nuevo registro.');
          return;
        }

        fillForm(data);
        showStatus('ok', 'Cliente encontrado. Revisa o agrega observaciones antes de guardar.');
      } catch (error) {
        form.elements.cliente_id.value = id;
        updateSummary();
        showStatus('bad', 'No se pudo buscar el ID. Revisa el webhook de busqueda en n8n.');
      } finally {
        buscarBtn.disabled = false;
        buscarBtn.textContent = 'Buscar ID';
      }
    });

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      showStatus('', '');

      const data = new FormData(form);
      data.set('origen', 'landing_control_interno');

      const buttons = Array.from(form.querySelectorAll('button[type="submit"]'));
      buttons.forEach((button) => {
        button.disabled = true;
        button.dataset.originalText = button.textContent;
        button.textContent = 'Guardando...';
      });

      try {
        const response = await fetch(WEBHOOK_FLUJO_4, {
          method: 'POST',
          body: data
        });

        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const result = await response.json().catch(() => ({}));

        showGeneratedLink(result);
      } catch (error) {
        showStatus('bad', `No se pudo enviar al Flujo 4: ${error.message}. Revisa webhook, CORS y mapeo a Selecciones.`);
      } finally {
        buttons.forEach((button) => {
          button.disabled = false;
          button.textContent = button.dataset.originalText || 'Guardar y generar link';
          delete button.dataset.originalText;
        });
      }
    });

    updateSummary();
  </script>
